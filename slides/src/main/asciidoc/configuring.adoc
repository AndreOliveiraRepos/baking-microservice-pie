== Docker-compose

[source,yaml]
----
- name: Deploy Book API
docker_container:
  name: book-api
  image: docker-registry:5000/baking/book-api:rpi
  state: started
  restart: yes
  pull: true
  ports:
   - "8081:8081"
  env:
    NUMBER_API_HOST: http://pi-grom-server-01:8084
----
